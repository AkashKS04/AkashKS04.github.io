Digital Thermometer

Built using Arduino Uno, LM35 sensor, and LCD to monitor real-time temperature.

View Details 
#include 
LiquidCrystal lcd(12, 11, 5, 4, 3, 2);
const int sensorPin = A0;
void setup() {
  lcd.begin(16, 2);
}
void loop() {
  float voltage = analogRead(sensorPin) * 5.0 / 1024.0;
  float tempC = voltage * 100.0;
  lcd.setCursor(0,0);
  lcd.print("Temp: ");
  lcd.print(tempC);
  lcd.print(" C");
  delay(1000);
}
